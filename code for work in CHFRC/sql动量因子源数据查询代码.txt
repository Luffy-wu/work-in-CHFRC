SELECT [ASHAREEODDERIVATIVEINDICATOR].[S_INFO_WINDCODE],
[ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT],
[ASHAREEODPRICES].[S_DQ_PCTCHANGE] AS [STOCK_PCTCHANGE],
[AINDEXEODPRICES].[S_DQ_PCTCHANGE] AS [INDEX_PCTCHANGE],
[ASHAREEODDERIVATIVEINDICATOR].[S_DQ_FREETURNOVER],
[ASHAREEODDERIVATIVEINDICATOR].[S_DQ_MV]
FROM [windfilesync].[dbo].[ASHAREEODDERIVATIVEINDICATOR],[windfilesync].[dbo].[ASHAREEODPRICES],[windfilesync].[dbo].[AINDEXEODPRICES]
WHERE [ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT] >= '20000101' 
AND [ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT] <= '20161231'
AND [ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT]=[ASHAREEODPRICES].[TRADE_DT]
AND [ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT]=[AINDEXEODPRICES].[TRADE_DT]
AND [ASHAREEODDERIVATIVEINDICATOR].[S_INFO_WINDCODE]=[ASHAREEODPRICES].[S_INFO_WINDCODE]
AND [AINDEXEODPRICES].[S_INFO_WINDCODE] = '000001.SH'
ORDER BY [ASHAREEODDERIVATIVEINDICATOR].[TRADE_DT],[ASHAREEODDERIVATIVEINDICATOR].[S_INFO_WINDCODE]
